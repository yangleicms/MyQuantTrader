FILE(GLOB_RECURSE SOURCE_FILES *.cpp)

 INCLUDE_DIRECTORIES(

)

${PROJECT_SOURCE_DIR}/lib/jsoncpp-1.8.3/lib
)

ADD_LIBRARY(FastAPI ${SOURCE_FILES})
